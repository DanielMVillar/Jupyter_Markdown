Técnicas con String¶

In [43]:
frase = "Mi primer cuaderno de Data Science con Python"
print(frase.upper()) #pone todas las letras en mayúsculas
print(frase.lower()) #pone todas las letras en minúsculas
print(frase.count('a')) #cuenta el número de carácteres definido entre paréntesis
print(frase.find('a')) #primera posición donde aparece el carácter definido entre paréntesis
print(frase.replace('a','e')) #sustituye el primer carácter por el segundo
print(frase.strip('con Python')) #elimina del principio y final los carácteres definidos entre paréntesis
print(frase.split(' ')) #separa el string por el carácter definido entre paréntesis y los almacena en una lista
